export function ProjectStats() {
  const [stats, setStats] = useState({ repos: 0, stars: 0, languages: 0 })
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    const fetchStats = async () => {
      try {
        const response = await fetch('/api/github-repos')
        if (response.ok) {
          const repos = await response.json()
          const totalStars = repos.reduce((sum, repo) => sum + repo.stargazers_count, 0)
          const uniqueLanguages = new Set(repos.filter(repo => repo.language).map(repo => repo.language))
          
          setStats({
            repos: repos.length,
            stars: totalStars,
            languages: uniqueLanguages.size
          })
        }
      } catch (error) {
        console.error('Failed to fetch stats:', error)
      } finally {
        setLoading(false)
      }
    }

    fetchStats()
  }, [])

  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ delay: 1.1, duration: 0.6 }}
      className="flex justify-center space-x-8 text-center mb-8"
    >
      <div className="flex flex-col items-center">
        <div className="text-3xl font-bold bg-gradient-to-r from-blue-600 to-purple-600 bg-clip-text text-transparent">
          {loading ? '...' : stats.repos}
        </div>
        <div className="text-sm text-gray-600 dark:text-gray-400">é¡¹ç›®</div>
      </div>
      <div className="flex flex-col items-center">
        <div className="text-3xl font-bold bg-gradient-to-r from-green-600 to-emerald-600 bg-clip-text text-transparent">
          {loading ? '...' : stats.stars}
        </div>
        <div className="text-sm text-gray-600 dark:text-gray-400">æ˜Ÿæ ‡</div>
      </div>
      <div className="flex flex-col items-center">
        <div className="text-3xl font-bold bg-gradient-to-r from-orange-600 to-red-600 bg-clip-text text-transparent">
          {loading ? '...' : stats.languages}
        </div>
        <div className="text-sm text-gray-600 dark:text-gray-400">è¯­è¨€</div>
      </div>
    </motion.div>
  )
}
